    vector<int> moveZeros(vector<int>& arr){
        vector<int> temp;
        for (int i = 0; i < arr.size(); i++){
            temp.push_back(arr[i]);
        }
        for (int i = 0; i < temp.size(); i++){
            arr[i] = temp[i];
        }
        for (int i = temp.size(); i < arr.size(); i++){
            arr[i] = 0;
        }
        return arr;
    }
